Items from the NLAâ€™s oral history collection can be found in Troveâ€™s Music, Audio, & Video category. def split_and_clean(value):
    values = value.split(" | ")
    return list(set([re.sub(r"(\w)--(\w)", r"\1 -- \2", v).strip(".") for v in values if v]))
